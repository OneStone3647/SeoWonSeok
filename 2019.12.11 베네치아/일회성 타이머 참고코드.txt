#include<iostream>	
#include<conio.h>
#include<time.h>
using namespace std;


void main()
{
	bool Status = false;
	int ExitTimer;
	while (true)
	{
		if (Status)
		{
			printf("%d\n", clock() - ExitTimer);
			if (clock() - ExitTimer >= 2000)
			{
				Status = false;
			}
		}
		if (kbhit())
		{
			char ch = getch();
			if (Status)
				continue;
			if (ch == 32)
			{
				ExitTimer = clock();
				Status = true;
			}
		}
	}
}
